CREATE DATABASE VOTE_DB;

-- 創建VOTEITEM表格
CREATE TABLE `VOTE_ITEM` (
    `ID` INT AUTO_INCREMENT PRIMARY KEY COMMENT '投票項目ID',
    `NAME` VARCHAR(50) NOT NULL COMMENT '投票項目',
    `VOTE_COUNT` INT NOT NULL DEFAULT 0 COMMENT '票數'
) ENGINE=INNODB DEFAULT CHARSET=UTF8MB4 COMMENT='投票項目表';

-- 創建VOTERECORD表格
CREATE TABLE `VOTE_RECORD` (
    `ID` INT AUTO_INCREMENT PRIMARY KEY COMMENT '投票記錄ID',
    `VOTER_NAME` VARCHAR(50) NOT NULL COMMENT '投票者',
    `VOTE_ITEM_ID` INT NOT NULL COMMENT '投票項目ID'
) ENGINE=INNODB DEFAULT CHARSET=UTF8MB4 COMMENT='投票記錄表';
